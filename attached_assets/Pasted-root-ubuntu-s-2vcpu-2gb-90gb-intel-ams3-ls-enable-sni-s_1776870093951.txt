#!/usr/bin/env bash
set -euo pipefail

CONF=/etc/nginx/sites-available/shopthelook
BACKUP=/root/shopthelook.nginx.bak.$(date +%s)

cp -a "$CONF" "$BACKUP"
echo ">>> Backed up current config to $BACKUP"

# Make the HTTPS server block the default for :443 so any SNI lands on it.
# Idempotent: only changes the line if it doesn't already say default_server.
sed -i 's|^\(\s*\)listen 443 ssl http2;|\1listen 443 ssl http2 default_server;|' "$CONF"
sed -i 's|^\(\s*\)listen \[::\]:443 ssl http2;|\1listen [::]:443 ssl http2 default_server;|' "$CONF"

# Also make :80 default_server so port-80 SNI tricks (rare) still redirect.
sed -i '0,/^\s*listen 80;/{s|^\(\s*\)listen 80;|\1listen 80 default_server;|}' "$CONF"
sed -i '0,/^\s*listen \[::\]:80;/{s|^\(\s*\)listen \[::\]:80;|\1listen [::]:80 default_server;|}' "$CONF"

echo ">>> Patched directives:"
grep -nE 'listen (80|443|\[::\]:(80|443))' "$CONF"

# Validate; restore on failure.
if ! nginx -t; then
  echo ">>> nginx -t FAILED — restoring backup."
  cp -a "$BACKUP" "$CONF"
  nginx -t
  exit 1
fi

systemctl reload nginx
echo ">>> nginx reloaded."

echo
echo ">>> Proof: WS upgrade succeeds even when SNI = myaccount.telkom.co.ke"
curl -sk -o /dev/null -D - --http1.1 \
